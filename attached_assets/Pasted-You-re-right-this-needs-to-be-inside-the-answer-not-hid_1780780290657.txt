I searched current OpenClaw docs plus major agentic frameworks and built the tool pack below on top of the runtime shape you already had. Your prior runtime already had the correct skeleton: `ToolDef/ToolSpec`, registry, execution, verification, memory, and audit flow. 

OpenClaw’s docs describe tools as typed callable actions such as `exec`, `browser`, `web_search`, `message`, and `image_generate`; skills are instruction packs; plugins add runtime capability and can register tools. OpenClaw’s own tool groups include runtime, filesystem, sessions, memory, web, UI/browser, automation, messaging, nodes, agents, media, OpenClaw built-ins, and plugin/MCP tools. ([OpenClaw][1]) AutoGPT’s built-in command providers include `finish`, `ask_user`, file tools, shell/Python execution, git clone, image generation, web search, website reading, and context open/close tools. ([GitHub][2]) LangChain, AutoGen, OpenAI Agents SDK, and MCP all converge on the same core idea: tools are callable functions with defined input/output schemas, and the runtime executes them outside the model. ([LangChain Docs][3])

Important truth: there is no honest way to paste **every OpenClaw ecosystem skill** because the community registry is huge and dynamic. One OpenClaw skill collection says it filtered **5,400+ skills** from ClawHub and excluded thousands more as spam, duplicates, low-quality, finance/crypto, or malicious-risk categories. ([GitHub][4]) So below is the correct move: a **universal tool-calling catalog** with the OpenClaw first-class tools, AutoGPT-style tools, MCP/plugin hooks, and the major agentic runtime functions you actually need.

JSONSchema = Dict[str, Any]
ToolHandler = Callable[[Dict[str, Any], "ToolContext"], Dict[str, Any]]


@dataclass
class ToolDef:
    name: str
    category: str
    risk: str
    description: str
    input_schema: JSONSchema = field(default_factory=dict)
    output_schema: JSONSchema = field(default_factory=dict)
    requires_auth: bool = False
    requires_approval: bool = False
    enabled_by_default: bool = False
    source_family: str = "universal"


@dataclass
class ToolContext:
    workspace: Path
    allow_network: bool = False
    allow_shell: bool = False
    allow_destructive: bool = False
    env: Dict[str, str] = field(default_factory=dict)
    memory: Dict[str, Any] = field(default_factory=dict)
    audit: List[Dict[str, Any]] = field(default_factory=list)

    def log(self, event_type: str, payload: Dict[str, Any]) -> None:
        self.audit.append({
            "ts": int(time.time()),
            "event_type": event_type,
            "payload": payload,
        })


def obj_schema(properties: Dict[str, Any], required: Optional[List[str]] = None) -> JSONSchema:
    return {
        "type": "object",
        "properties": properties,
        "required": required or [],
        "additionalProperties": False,
    }


def string_prop(description: str, default: Optional[str] = None) -> Dict[str, Any]:
    prop: Dict[str, Any] = {"type": "string", "description": description}
    if default is not None:
        prop["default"] = default
    return prop


def bool_prop(description: str, default: Optional[bool] = None) -> Dict[str, Any]:
    prop: Dict[str, Any] = {"type": "boolean", "description": description}
    if default is not None:
        prop["default"] = default
    return prop


def int_prop(description: str, default: Optional[int] = None, minimum: Optional[int] = None) -> Dict[str, Any]:
    prop: Dict[str, Any] = {"type": "integer", "description": description}
    if default is not None:
        prop["default"] = default
    if minimum is not None:
        prop["minimum"] = minimum
    return prop


def array_prop(description: str, item_schema: Dict[str, Any]) -> Dict[str, Any]:
    return {"type": "array", "description": description, "items": item_schema}


def result_schema() -> JSONSchema:
    return obj_schema({
        "ok": {"type": "boolean"},
        "data": {"description": "Tool-specific result payload"},
        "message": {"type": "string"},
        "error": {"type": ["string", "null"]},
    }, ["ok"])

class ToolRegistry:
    def __init__(self, tool_defs: Optional[List[ToolDef]] = None):
        self.tool_defs: Dict[str, ToolDef] = {t.name: t for t in (tool_defs or TOOL_DEFS)}
        self.handlers: Dict[str, ToolHandler] = {}

    def register_handler(self, name: str, handler: ToolHandler) -> None:
        if name not in self.tool_defs:
            raise KeyError(f"Unknown tool: {name}")
        self.handlers[name] = handler

    def register_default_handlers(self) -> None:
        for name in ["read", "read_file", "open_file"]:
            self.register_handler(name, handle_read)
        for name in ["write", "write_file"]:
            self.register_handler(name, handle_write)
        for name in ["list_folder", "list_directory", "open_folder"]:
            self.register_handler(name, handle_list_directory)

        self.register_handler("edit", handle_edit)
        self.register_handler("file_exists", handle_file_exists)
        self.register_handler("search_files", handle_search_files)
        self.register_handler("grep_files", handle_grep_files)
        self.register_handler("make_directory", handle_make_directory)
        self.register_handler("delete_path", handle_delete_path)
        self.register_handler("calculator", handle_calculator)

        self.register_handler("memory_get", handle_memory_get)
        self.register_handler("memory_put", handle_memory_put)
        self.register_handler("memory_search", handle_memory_search)

        self.register_handler("tool_search", self.handle_tool_search)
        self.register_handler("tool_search_code", self.handle_tool_search_code)
        self.register_handler("tool_describe", self.handle_tool_describe)

        self.register_handler("finish", handle_finish)
        self.register_handler("ask_user", handle_ask_user)
        self.register_handler("update_plan", handle_update_plan)
        self.register_handler("session_status", handle_session_status)

        for name in [
            "exec",
            "bash",
            "execute_shell",
            "execute_shell_popen",
            "execute_python_code",
            "execute_python_file",
            "run_tests",
            "run_build",
            "git_status",
            "git_diff",
        ]:
            self.register_handler(name, handle_shell_gated)

        for name in self.tool_defs:
            if name not in self.handlers:
                self.handlers[name] = handle_not_configured

    def list_tools(self, category: Optional[str] = None, enabled_only: bool = False) -> List[Dict[str, Any]]:
        tools = list(self.tool_defs.values())
        if category:
            tools = [t for t in tools if t.category == category]
        if enabled_only:
            tools = [t for t in tools if t.enabled_by_default]
        return [asdict(t) for t in tools]

    def describe(self, name: str) -> Dict[str, Any]:
        tool = self.tool_defs.get(name)
        if not tool:
            return {"ok": False, "error": "unknown_tool", "message": f"No tool named {name}"}
        return {"ok": True, "data": asdict(tool)}

    def invoke(self, name: str, args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
        tool = self.tool_defs.get(name)
        if not tool:
            return {"ok": False, "error": "unknown_tool", "message": f"No tool named {name}"}

        if tool.risk == "destructive" and not ctx.allow_destructive:
            return {"ok": False, "error": "destructive_tool_blocked", "message": f"{name} requires destructive approval."}

        if tool.category == "runtime" and tool.risk in {"high", "destructive"} and not ctx.allow_shell:
            return {"ok": False, "error": "shell_blocked", "message": f"{name} requires allow_shell=True."}

        if tool.category in {"web", "api"} and not ctx.allow_network:
            return {"ok": False, "error": "network_blocked", "message": f"{name} requires allow_network=True."}

        ctx.log("tool_call", {"name": name, "args": safe_log_args(args), "risk": tool.risk, "category": tool.category})
        result = self.handlers.get(name, handle_not_configured)(args, ctx)
        ctx.log("tool_result", {"name": name, "ok": result.get("ok"), "error": result.get("error")})
        return result

    def handle_tool_search(self, args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
        query = str(args.get("query", "")).lower()
        category = args.get("category")
        matches = []
        for tool in self.tool_defs.values():
            haystack = f"{tool.name} {tool.category} {tool.description} {tool.source_family}".lower()
            if query in haystack and (not category or category == tool.category):
                matches.append(asdict(tool))
        return {"ok": True, "data": matches, "message": f"Found {len(matches)} tools."}

    def handle_tool_search_code(self, args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
        query = str(args.get("query", "")).lower()
        language = args.get("language", "json")
        matches = []
        for tool in self.tool_defs.values():
            haystack = f"{tool.name} {tool.category} {tool.description} {tool.source_family}".lower()
            if query in haystack:
                if language == "python":
                    matches.append(f"def {tool.name}(args: dict) -> dict: ...")
                elif language == "typescript":
                    matches.append(f"async function {tool.name}(args: Record<string, unknown>): Promise<ToolResult> {{ /* ... */ }}")
                else:
                    matches.append(asdict(tool))
        return {"ok": True, "data": matches, "message": f"Found {len(matches)} signatures."}

    def handle_tool_describe(self, args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
        return self.describe(str(args.get("name", "")))

def safe_log_args(args: Dict[str, Any]) -> Dict[str, Any]:
    redacted = {}
    for key, value in args.items():
        if re.search(r"token|secret|password|api[_-]?key|authorization", key, re.I):
            redacted[key] = "***REDACTED***"
        else:
            redacted[key] = value
    return redacted


def resolve_workspace_path(ctx: ToolContext, raw_path: str) -> Path:
    base = ctx.workspace.resolve()
    target = (base / raw_path).resolve()
    if not str(target).startswith(str(base)):
        raise PermissionError(f"Path escapes workspace: {raw_path}")
    return target


def handle_read(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args["path"])
    if not path.exists():
        return {"ok": False, "error": "not_found", "message": str(path)}
    if not path.is_file():
        return {"ok": False, "error": "not_file", "message": str(path)}
    return {"ok": True, "data": path.read_text(encoding="utf-8", errors="replace"), "message": str(path)}


def handle_write(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args["path"])
    overwrite = bool(args.get("overwrite", False))
    if path.exists() and not overwrite:
        return {"ok": False, "error": "exists", "message": "Set overwrite=true to replace existing file."}
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(str(args["content"]), encoding="utf-8")
    return {"ok": True, "data": {"path": str(path), "bytes": path.stat().st_size}, "message": "written"}


def handle_edit(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args["path"])
    text = path.read_text(encoding="utf-8", errors="replace")
    old = str(args["old_text"])
    new = str(args["new_text"])
    if old not in text:
        return {"ok": False, "error": "old_text_not_found", "message": "Exact text not found."}
    updated = text.replace(old, new, 1)
    path.write_text(updated, encoding="utf-8")
    return {"ok": True, "data": {"path": str(path), "replacements": 1}, "message": "edited"}


def handle_list_directory(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args.get("path", "."))
    if not path.exists():
        return {"ok": False, "error": "not_found", "message": str(path)}
    if not path.is_dir():
        return {"ok": False, "error": "not_directory", "message": str(path)}
    items = []
    for p in sorted(path.iterdir()):
        items.append({"name": p.name, "type": "dir" if p.is_dir() else "file", "size": p.stat().st_size})
    return {"ok": True, "data": items, "message": f"{len(items)} items"}


def handle_file_exists(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args["path"])
    return {"ok": True, "data": {"exists": path.exists(), "is_file": path.is_file(), "is_dir": path.is_dir()}, "message": str(path)}


def handle_search_files(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    root = resolve_workspace_path(ctx, args.get("path", "."))
    pattern = args["pattern"]
    matches = [str(p.relative_to(ctx.workspace.resolve())) for p in root.glob(pattern)]
    return {"ok": True, "data": matches, "message": f"{len(matches)} matches"}


def handle_grep_files(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    root = resolve_workspace_path(ctx, args.get("path", "."))
    regex = re.compile(args["pattern"])
    glob_pat = args.get("glob", "**/*")
    max_matches = int(args.get("max_matches", 100))
    matches = []
    for p in root.glob(glob_pat):
        if not p.is_file():
            continue
        try:
            text = p.read_text(encoding="utf-8", errors="replace")
        except Exception:
            continue
        for lineno, line in enumerate(text.splitlines(), start=1):
            if regex.search(line):
                matches.append({"path": str(p.relative_to(ctx.workspace.resolve())), "line": lineno, "text": line[:500]})
                if len(matches) >= max_matches:
                    return {"ok": True, "data": matches, "message": f"{len(matches)} matches"}
    return {"ok": True, "data": matches, "message": f"{len(matches)} matches"}


def handle_make_directory(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    path = resolve_workspace_path(ctx, args["path"])
    path.mkdir(parents=True, exist_ok=True)
    return {"ok": True, "data": {"path": str(path)}, "message": "directory_created"}


def handle_delete_path(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    if not ctx.allow_destructive:
        return {"ok": False, "error": "destructive_blocked", "message": "Deletion requires allow_destructive=True."}
    path = resolve_workspace_path(ctx, args["path"])
    if not path.exists():
        return {"ok": False, "error": "not_found", "message": str(path)}
    if path.is_dir():
        path.rmdir()
    else:
        path.unlink()
    return {"ok": True, "data": {"path": str(path)}, "message": "deleted"}


def handle_calculator(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    expr = str(args["expression"])
    if not re.fullmatch(r"[0-9+\-*/(). %]+", expr):
        return {"ok": False, "error": "unsafe_expression", "message": "Only arithmetic characters are allowed."}
    try:
        value = eval(expr, {"__builtins__": {}}, {})
    except Exception as exc:
        return {"ok": False, "error": exc.__class__.__name__, "message": str(exc)}
    return {"ok": True, "data": value, "message": str(value)}


def handle_memory_get(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:

    return {"ok": True, "data": ctx.memory.get(key), "message": key}


def handle_memory_put(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:

    ctx.memory[key] = args.get("value")
    return {"ok": True, "data": {"key": key, "saved": True}, "message": "saved"}


def handle_memory_search(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    query = str(args["query"]).lower()
    limit = int(args.get("limit", 5))
    results = []
    for key, value in ctx.memory.items():
        haystack = f"{key} {json.dumps(value, ensure_ascii=False)}".lower()
        if query in haystack:
            results.append({"key": key, "value": value})
            if len(results) >= limit:
                break
    return {"ok": True, "data": results, "message": f"{len(results)} matches"}


def handle_shell_gated(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    if not ctx.allow_shell:
        return {"ok": False, "error": "shell_blocked", "message": "Shell execution requires allow_shell=True."}

    command = args.get("command")
    if not command and "path" in args:
        command = f"python {args['path']}"
        if args.get("args"):
            command += " " + " ".join(map(str, args["args"]))
    if not command and "code" in args:
        tmp = ctx.workspace / f".agent_tmp_{uuid.uuid4().hex}.py"
        tmp.write_text(str(args["code"]), encoding="utf-8")
        command = f"python {tmp.name}"

    timeout = int(args.get("timeout_sec", 30))

    try:
        proc = subprocess.run(
            command,
            cwd=str(ctx.workspace),
            shell=True,
            capture_output=True,
            text=True,
            timeout=timeout,
            env={**os.environ, **ctx.env},
        )
    except subprocess.TimeoutExpired:
        return {"ok": False, "error": "timeout", "message": f"Timed out after {timeout}s"}

    return {
        "ok": proc.returncode == 0,
        "data": {"stdout": proc.stdout, "stderr": proc.stderr, "returncode": proc.returncode},
        "message": "command_completed" if proc.returncode == 0 else "command_failed",
        "error": None if proc.returncode == 0 else "nonzero_exit",
    }


def handle_finish(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    return {"ok": True, "data": {"final": args["answer"]}, "message": "finished"}


def handle_ask_user(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:

    return {"ok": True, "data": {"workspace": str(ctx.workspace), "audit_events": len(ctx.audit)}, "message": "ok"}


def handle_not_configured(args: Dict[str, Any], ctx: ToolContext) -> Dict[str, Any]:
    return {
        "ok": False,
        "error": "not_configured",
        "message": "This tool is declared but not wired. Add a real handler with credentials, policy, and verification.",
        "data": {"args_received": safe_log_args(args)},
    }

function safeLogArgs(args: Record<string, unknown>): Record<string, unknown> {
  const out: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(args)) {
    if (/token|secret|password|api[_-]?key|authorization/i.test(key)) {
      out[key] = "***REDACTED***";
    } else {
      out[key] = value;
    }
  }
  return out;
}

function resolveWorkspacePath(ctx: ToolContext, rawPath: string): string {
  const base = path.resolve(ctx.workspace);
  const target = path.resolve(base, rawPath);
  if (!target.startsWith(base)) {
    throw new Error(`Path escapes workspace: ${rawPath}`);
  }
  return target;
}

function ok(data?: unknown, message = "ok"): ToolResult {
  return { ok: true, data, message, error: null };
}

function fail(error: string, message: string, data?: unknown): ToolResult {
  return { ok: false, error, message, data };
}


export class ToolRegistry {
  private toolDefs: Map<string, ToolDef>;
  private handlers: Map<string, ToolHandler>;

  constructor(toolDefs: ToolDef[] = TOOL_DEFS) {
    this.toolDefs = new Map(toolDefs.map((t) => [t.name, t]));
    this.handlers = new Map();
  }

  registerHandler(name: string, handler: ToolHandler): void {
    if (!this.toolDefs.has(name)) {
      throw new Error(`Unknown tool: ${name}`);
    }
    this.handlers.set(name, handler);
  }

  registerDefaultHandlers(): void {
    for (const name of ["read", "read_file", "open_file"]) this.registerHandler(name, handleRead);
    for (const name of ["write", "write_file"]) this.registerHandler(name, handleWrite);
    for (const name of ["list_folder", "list_directory", "open_folder"]) this.registerHandler(name, handleListDirectory);

    this.registerHandler("edit", handleEdit);
    this.registerHandler("file_exists", handleFileExists);
    this.registerHandler("search_files", handleSearchFiles);
    this.registerHandler("grep_files", handleGrepFiles);
    this.registerHandler("make_directory", handleMakeDirectory);
    this.registerHandler("delete_path", handleDeletePath);
    this.registerHandler("calculator", handleCalculator);

    this.registerHandler("memory_get", handleMemoryGet);
    this.registerHandler("memory_put", handleMemoryPut);
    this.registerHandler("memory_search", handleMemorySearch);

    this.registerHandler("tool_search", (args, ctx) => this.handleToolSearch(args, ctx));
    this.registerHandler("tool_search_code", (args, ctx) => this.handleToolSearchCode(args, ctx));
    this.registerHandler("tool_describe", (args, ctx) => this.handleToolDescribe(args, ctx));

    this.registerHandler("finish", handleFinish);
    this.registerHandler("ask_user", handleAskUser);
    this.registerHandler("update_plan", handleUpdatePlan);
    this.registerHandler("session_status", handleSessionStatus);

    for (const name of ["exec", "bash", "execute_shell", "execute_shell_popen", "execute_python_code", "execute_python_file", "run_tests", "run_build", "git_status", "git_diff"]) {
      this.registerHandler(name, handleShellGated);
    }

    for (const name of this.toolDefs.keys()) {
      if (!this.handlers.has(name)) {
        this.handlers.set(name, handleNotConfigured);
      }
    }
  }

  listTools(category?: ToolCategory, enabledOnly = false): ToolDef[] {
    let tools = Array.from(this.toolDefs.values());
    if (category) tools = tools.filter((t) => t.category === category);
    if (enabledOnly) tools = tools.filter((t) => t.enabledByDefault);
    return tools;
  }

  describe(name: string): ToolResult {
    const toolDef = this.toolDefs.get(name);
    if (!toolDef) return fail("unknown_tool", `No tool named ${name}`);
    return ok(toolDef);
  }

  async invoke(name: string, args: Record<string, unknown>, ctx: ToolContext): Promise<ToolResult> {
    const toolDef = this.toolDefs.get(name);
    if (!toolDef) return fail("unknown_tool", `No tool named ${name}`);

    if (toolDef.risk === "destructive" && !ctx.allowDestructive) {
      return fail("destructive_tool_blocked", `${name} requires destructive approval.`);
    }

    if (toolDef.category === "runtime" && ["high", "destructive"].includes(toolDef.risk) && !ctx.allowShell) {
      return fail("shell_blocked", `${name} requires allowShell=true.`);
    }

    if (["web", "api"].includes(toolDef.category) && !ctx.allowNetwork) {
      return fail("network_blocked", `${name} requires allowNetwork=true.`);
    }

    ctx.audit.push({
      ts: Date.now(),
      event_type: "tool_call",
      payload: {
        name,
        args: safeLogArgs(args),
        risk: toolDef.risk,
        category: toolDef.category,
      },
    });

    const handler = this.handlers.get(name) || handleNotConfigured;
    const result = await handler(args, ctx);

    ctx.audit.push({
      ts: Date.now(),
      event_type: "tool_result",
      payload: {
        name,
        ok: result.ok,
        error: result.error,
      },
    });

    return result;
  }

  private handleToolSearch(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
    const query = String(args.query || "").toLowerCase();
    const category = args.category ? String(args.category) : undefined;
    const matches = Array.from(this.toolDefs.values()).filter((t) => {
      const haystack = `${t.name} ${t.category} ${t.description} ${t.sourceFamily}`.toLowerCase();
      return haystack.includes(query) && (!category || t.category === category);
    });
    return ok(matches, `Found ${matches.length} tools.`);
  }

  private handleToolSearchCode(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
    const query = String(args.query || "").toLowerCase();
    const language = String(args.language || "json");
    const matches = Array.from(this.toolDefs.values())
      .filter((t) => `${t.name} ${t.category} ${t.description} ${t.sourceFamily}`.toLowerCase().includes(query))
      .map((t) => {
        if (language === "python") return `def ${t.name}(args: dict) -> dict: ...`;
        if (language === "typescript") return `async function ${t.name}(args: Record<string, unknown>): Promise<ToolResult> { /* ... */ }`;
        return t;
      });
    return ok(matches, `Found ${matches.length} signatures.`);
  }

  private handleToolDescribe(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
    return this.describe(String(args.name || ""));
  }
}


function handleRead(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const filePath = resolveWorkspacePath(ctx, String(args.path));
  if (!fs.existsSync(filePath)) return fail("not_found", filePath);
  if (!fs.statSync(filePath).isFile()) return fail("not_file", filePath);
  return ok(fs.readFileSync(filePath, "utf-8"), filePath);
}

function handleWrite(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const filePath = resolveWorkspacePath(ctx, String(args.path));
  const overwrite = Boolean(args.overwrite);
  if (fs.existsSync(filePath) && !overwrite) {
    return fail("exists", "Set overwrite=true to replace existing file.");
  }
  fs.mkdirSync(path.dirname(filePath), { recursive: true });
  fs.writeFileSync(filePath, String(args.content || ""), "utf-8");
  return ok({ path: filePath, bytes: fs.statSync(filePath).size }, "written");
}

function handleEdit(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const filePath = resolveWorkspacePath(ctx, String(args.path));
  const oldText = String(args.old_text);
  const newText = String(args.new_text);
  const text = fs.readFileSync(filePath, "utf-8");
  if (!text.includes(oldText)) return fail("old_text_not_found", "Exact text not found.");
  fs.writeFileSync(filePath, text.replace(oldText, newText), "utf-8");
  return ok({ path: filePath, replacements: 1 }, "edited");
}

function handleListDirectory(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const dirPath = resolveWorkspacePath(ctx, String(args.path || "."));
  if (!fs.existsSync(dirPath)) return fail("not_found", dirPath);
  if (!fs.statSync(dirPath).isDirectory()) return fail("not_directory", dirPath);

  const items = fs.readdirSync(dirPath).map((name) => {
    const full = path.join(dirPath, name);
    const stat = fs.statSync(full);
    return {
      name,
      type: stat.isDirectory() ? "dir" : "file",
      size: stat.size,
    };
  });

  return ok(items, `${items.length} items`);
}

function handleFileExists(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const target = resolveWorkspacePath(ctx, String(args.path));
  const exists = fs.existsSync(target);
  return ok({
    exists,
    is_file: exists ? fs.statSync(target).isFile() : false,
    is_dir: exists ? fs.statSync(target).isDirectory() : false,
  }, target);
}

function walkFiles(root: string): string[] {
  const out: string[] = [];

  for (const name of fs.readdirSync(root)) {
    const full = path.join(root, name);
    const stat = fs.statSync(full);
    if (stat.isDirectory()) out.push(...walkFiles(full));
    else out.push(full);
  }

  return out;
}

function handleSearchFiles(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const root = resolveWorkspacePath(ctx, String(args.path || "."));
  const pattern = String(args.pattern || "").replace(/\*/g, "");
  const files = walkFiles(root).filter((file) => file.includes(pattern));
  return ok(files.map((f) => path.relative(ctx.workspace, f)), `${files.length} matches`);
}

function handleGrepFiles(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const root = resolveWorkspacePath(ctx, String(args.path || "."));
  const regex = new RegExp(String(args.pattern));
  const max = Number(args.max_matches || 100);
  const matches: Array<Record<string, unknown>> = [];

  for (const file of walkFiles(root)) {
    const text = fs.readFileSync(file, "utf-8");
    const lines = text.split(/\r?\n/);
    for (let idx = 0; idx < lines.length; idx++) {
      if (regex.test(lines[idx])) {
        matches.push({
          path: path.relative(ctx.workspace, file),
          line: idx + 1,
          text: lines[idx].slice(0, 500),
        });
        if (matches.length >= max) return ok(matches, `${matches.length} matches`);
      }
    }
  }

  return ok(matches, `${matches.length} matches`);
}

function handleMakeDirectory(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const dirPath = resolveWorkspacePath(ctx, String(args.path));
  fs.mkdirSync(dirPath, { recursive: true });
  return ok({ path: dirPath }, "directory_created");
}

function handleDeletePath(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  if (!ctx.allowDestructive) return fail("destructive_blocked", "Deletion requires allowDestructive=true.");
  const target = resolveWorkspacePath(ctx, String(args.path));
  if (!fs.existsSync(target)) return fail("not_found", target);
  const stat = fs.statSync(target);
  if (stat.isDirectory()) fs.rmdirSync(target);
  else fs.unlinkSync(target);
  return ok({ path: target }, "deleted");
}

function handleCalculator(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
  const expr = String(args.expression || "");
  if (!/^[0-9+\-*/(). %]+$/.test(expr)) {
    return fail("unsafe_expression", "Only arithmetic characters are allowed.");
  }

  try {
    const value = Function(`"use strict"; return (${expr});`)();
    return ok(value, String(value));
  } catch (err) {
    return fail("calculation_error", String(err));
  }
}

function handleMemoryGet(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const key = String(args.key);
  return ok(ctx.memory[key], key);
}

function handleMemoryPut(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const key = String(args.key);
  ctx.memory[key] = args.value;
  return ok({ key, saved: true }, "saved");
}

function handleMemorySearch(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  const query = String(args.query || "").toLowerCase();
  const limit = Number(args.limit || 5);
  const results: Array<Record<string, unknown>> = [];

  for (const [key, value] of Object.entries(ctx.memory)) {
    const haystack = `${key} ${JSON.stringify(value)}`.toLowerCase();
    if (haystack.includes(query)) {
      results.push({ key, value });
      if (results.length >= limit) break;
    }
  }

  return ok(results, `${results.length} matches`);
}

function handleShellGated(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  if (!ctx.allowShell) return fail("shell_blocked", "Shell execution requires allowShell=true.");

  let command = String(args.command || "");

  if (!command && args.path) {
    command = `python ${String(args.path)}`;
  }

  if (!command) return fail("missing_command", "No command provided.");

  const timeoutSec = Number(args.timeout_sec || 30);

  try {
    const output = childProcess.execSync(command, {
      cwd: ctx.workspace,
      timeout: timeoutSec * 1000,
      encoding: "utf-8",
      env: { ...process.env, ...ctx.env },
      stdio: "pipe",
    });

    return ok({ stdout: output, stderr: "", returncode: 0 }, "command_completed");
  } catch (err: any) {
    return {
      ok: false,
      error: "command_failed",
      message: err.message || "Command failed.",
      data: {
        stdout: err.stdout?.toString?.() || "",
        stderr: err.stderr?.toString?.() || "",
        returncode: err.status ?? 1,
      },
    };
  }
}

function handleFinish(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
  return ok({ final: args.answer }, "finished");
}

function handleAskUser(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
  return ok({ question: args.question, options: args.options || [] }, "user_input_required");
}

function handleUpdatePlan(args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  ctx.memory["_current_plan"] = args.steps;
  return ok({ steps: args.steps }, "plan_updated");
}

function handleSessionStatus(_args: Record<string, unknown>, ctx: ToolContext): ToolResult {
  return ok({ workspace: ctx.workspace, audit_events: ctx.audit.length }, "ok");
}

function handleNotConfigured(args: Record<string, unknown>, _ctx: ToolContext): ToolResult {
  return fail(
    "not_configured",
    "This tool is declared but not wired. Add a real handler with credentials, policy, and verification.",
    { args_received: safeLogArgs(args) }
  );
}

## Real or bullshit?

**Real:** this is the correct production shape.

**Bullshit:** calling external systems “done” before wiring the real handler, credentials, permissions, and verification.

The critical rule is this:

```text
A declared tool is not a working tool.
A wired handler is not a safe tool.
A safe tool is one with:
schema + auth + permission gate + sandbox + audit + verification.
```
